def _apply_orders(self, msg):
    assert self._depth_cache
    self._depth_cache._bids = msg.get("b", [])
    self._depth_cache._asks = msg.get("a", [])

    # keeping update time
    self._depth_cache.update_time = msg.get("E") or msg.get("lastUpdateId")
